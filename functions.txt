CREATE OR REPLACE FUNCTION TAX(S NUMBER)
RETURN NUMBER
IS
RESULT NUMBER(7,2);
BEGIN
RESULT := S * 21 / 100;
RETURN
END;

CREATE OR REPLACE FUNCTION DNAME (D NUMBER)
RETURN CHAR
IS
DN DEPARTMENTS.DEPARTMENT_NAME%TYPE;
BEGIN
SELECT DEPARTMENT_NAME INTO DN FROM DEPARTMENTS WHERE DEPARTMENT_ID = D;
RETURN DN;
END;
