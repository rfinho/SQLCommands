CREATE OR REPLACE FUNCTION TAX(S NUMBER)
RETURN NUMBER
IS
RESULT NUMBER(7,2);
BEGIN
RESULT := S * 21 / 100;
RETURN
END;

CREATE OR REPLACE FUNCTION DNAME (D NUMBER)
RETURN CHAR
IS
DN DEPARTMENTS.DEPARTMENT_NAME%TYPE;
BEGIN
SELECT DEPARTMENT_NAME INTO DN FROM DEPARTMENTS WHERE DEPARTMENT_ID = D;
RETURN DN;
END;

CREATE OR REPLACE FUNCTION COUNT_SUBCATEGORY(C NUMBER)
RETURN NUMBER
IS
RESULT NUMBER(3);
BEGIN
SELECT NVL(MAX(SID),0) INTO RESULT FROM SUBCATEGORY WHERE CID = C;
RETURN RESULT;
END;

CREATE OR REPLACE FUNCTION COUNT_PRODUCTS(C NUMBER)
RETURN NUMBER
IS
RESULT NUMBER(3);
BEGIN
SELECT NVL(MAX(PID),0) INTO RESULT FROM PRODUCTS WHERE SID IN(SELECT SID FROM SUBCATEGORY WHERE SID = C);
RETURN RESULT;
END;
