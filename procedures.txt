CREATE OR REPLACE PROCEDURE Add1(
 A Number,
 B Number,
 C Number)
 IS
 Result Number(3);
 BEGIN
 Result:= A+B+C;
 DBMS_output.put_line('Result is ' || Result);
 END;

  name char,
